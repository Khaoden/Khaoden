function toggleLanguage() {
  const zh = document.querySelector('[lang="zh"]');
  const en = document.querySelector('[lang="en"]');
  const button = document.querySelector('.lang-switch');
  
  if (en.classList.contains('active')) {
    en.classList.remove('active');
    zh.classList.add('active');
    button.style.background = 'linear-gradient(45deg, #FF4B2B 30%, #FF416C 90%)';
  } else {
    zh.classList.remove('active');
    en.classList.add('active');
    button.style.background = 'linear-gradient(45deg, #2196F3 30%, #21CBF3 90%)';
  }
  
  // 添加按钮点击特效
  button.style.transform = 'scale(0.95)';
  setTimeout(() => {
    button.style.transform = 'scale(1)';
  }, 100);
}

// 确保默认显示英文
document.addEventListener('DOMContentLoaded', function() {
  const zh = document.querySelector('[lang="zh"]');
  const en = document.querySelector('[lang="en"]');
  zh.classList.remove('active');
  en.classList.add('active');
});
</script>
